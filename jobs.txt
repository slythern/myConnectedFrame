# Turn off display every evening 
# CST 0 2 * * * echo 1 > /sys/class/backlight/rpi_backlight/bl_power
# CDT/MST
0 3 * * * echo 1 > /sys/class/backlight/rpi_backlight/bl_power
# MDT/PST 0 4 * * * echo 1 > /sys/class/backlight/rpi_backlight/bl_power
# PDT 0 5 * * * echo 1 > /sys/class/backlight/rpi_backlight/bl_power

# Turn on display every morning 
# CST 0 12 * * * echo 0 > /sys/class/backlight/rpi_backlight/bl_power
# CDT/MST
0 13 * * * echo 0 > /sys/class/backlight/rpi_backlight/bl_power
# MDT/PST 0 14 * * * echo 0 > /sys/class/backlight/rpi_backlight/bl_power
# PDT 0 15 * * * echo 0 > /sys/class/backlight/rpi_backlight/bl_power
